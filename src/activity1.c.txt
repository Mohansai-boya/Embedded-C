#include activity1.h


void peripheral_init(void)
{	
	DDRD = (1PD2);  set PD2=1 for LED
    DDRD &= ~(1PD0); clear bit
    PORTD = (1PD0); set bit PD0 for SeatSwitch
    DDRD &= ~(1PD1); clear bit
    PORTD = (1PD1); set bit PD0 for HeaterSwitch
}

void TurnLED_ON(){
    LED_PORT = (1LED_PIN); 
}

void TurnLED_OFF(){
    LED_PORT &= ~(1LED_PIN);
}

int act1=0;
int activity1_LED(void)
{
       peripheral_init();
        if(!(PIND&(1BUTTON_SENSOR )) && !(PIND&(1TEMP_SENSOR))) both the switches are pressed
        { 
            act1=1;
        }
        else  in all other cases
        {
            act1=0;
        }
    return act1;
}
